def bernoulliParameterMLE(data):
    return (1/data.label.count())*data[data.label==1].x.count()
def gaussianMeanMLE(data,i):
    return np.array(data[data.label==i][['x','y']].mean())
def covarianceMatrixMLE(data):
    S1=np.zeros((2,2))
    S2=np.zeros((2,2))
    data_centred_0=(data[data.label==0][['x','y']]-gaussianMeanMLE(data,0))
    data_centred_1=(data[data.label==1][['x','y']]-gaussianMeanMLE(data,1))
    for i in range (0,len(data_centred_0)):
        S1=S1+np.dot(np.array(data_centred_0.iloc[i]).reshape(2,1),np.array(data_centred_0.iloc[i]).reshape(1,2))
    for i in range (0,len(data_centred_1)):
        S2=S2+np.dot(np.array(data_centred_1.iloc[i]).reshape(2,1),np.array(data_centred_1.iloc[i]).reshape(1,2))
    return  (1/data.label.count())*(S1+S2)   
def w_LDA(data):
    return np.dot(np.linalg.inv(covarianceMatrixMLE(data)),gaussianMeanMLE(data,0)-gaussianMeanMLE(data,1))
def b_LDA(data):
    S1=0.5*(np.dot(gaussianMeanMLE(data,1),np.dot(np.linalg.inv(covarianceMatrixMLE(data)),gaussianMeanMLE(data,1))))
    S2=0.5*(np.dot(gaussianMeanMLE(data,0),np.dot(np.linalg.inv(covarianceMatrixMLE(data)),gaussianMeanMLE(data,0))))
    S3=np.log2((1-bernoulliParameterMLE(data))/bernoulliParameterMLE(data))
    return S1-S2+S3

def IRLS(data):
    data['bias']=1  #Adding the column of bias to the table
    data=data.reindex(columns=['bias','x','y','label']) #swaping the column 
                                                        #for adapted representation
    w=np.array([[0,0,0]]) #Initializing w=[bias,x1,x2]
    R=np.eye((data.count()[0])) #Initializing the diagonal matrix :
                                                 #(sigmoid*(1-sigmoid))
    while (np.linalg.det(R)!=0):   #the stop condition, means convergence of the parameters.
        X=np.array(data[['bias','x','y']]) #The matrix of data
        for j in range(0,np.shape(R)[0]): 
            R[j,j]=sigmoid(np.dot(w,X[j]))*(1-sigmoid(np.dot(w,X[j]))) #calculating the new 
                                                                    #diagonal elements of R
        nu=np.array([[sigmoid(np.dot(w,X[j])) for j in range(np.shape(X)[0])]]) #The matrix of
                                                                                #sigmoids
        
        z=np.dot(X,w.T)-np.dot(np.linalg.inv(R),(nu-np.array([data.label])).T )  
        w=np.dot(np.dot(np.linalg.inv(np.dot(X.T,np.dot(R,X))),np.dot(X.T,R)),z).T
    return w

def linearRegression_w(data):
    new_labels=[]
    for i in range(data.count()[0]):
        if (data.label.iloc[i]==0):
            new_labels=new_labels+[[0,1]]
        else:
            new_labels=new_labels+[[1,0]]
    data['bias']=1  #Adding the column of bias to the table
    data=data.reindex(columns=['bias','x','y','label']) 
    X=np.array(data[['bias','x','y']])
    w=np.dot(np.dot(np.linalg.inv(np.dot(X.T,X)),X.T),new_labels)
    return w.T

def misclassification_error(testdata,traindata,algo):
    counter=0
    if (algo=='LDA'): 
        w=w_LDA(traindata)
        b=b_LDA(traindata)
        for i in range(testdata.count()[0]):
            if((np.dot(w,[testdata.x.iloc[i],testdata.y.iloc[i]])+b>=0 and testdata.label.iloc[i]==1) or ((np.dot(w,[testdata.x.iloc[i],testdata.y.iloc[i]])+b<=0 and testdata.label.iloc[i]==0))) :
                counter+=1
    if (algo=='LogR'):
        w=IRLS(traindata)
        for i in range(testdata.count()[0]):
            if(((np.dot(w,[1,testdata.x.iloc[i],testdata.y.iloc[i]]))<=0 and testdata.label.iloc[i]==1) or ((np.dot(w,[1,testdata.x.iloc[i],testdata.y.iloc[i]])>=0 and testdata.label.iloc[i]==0))) :
                counter+=1  
    if (algo=='LinearR'):
        w=linearRegression_w(traindata)
        for i in range(testdata.count()[0]):
            x=testdata.x.iloc[i]
            y=testdata.y.iloc[i]
            if(((w[0,1]*x+w[0,2]*y+w[0,0]-w[1,1]*x-w[1,2]*y-w[1,0]<=0 and testdata.label.iloc[i]==1) or (w[0,1]*x+w[0,2]*y+w[0,0]-w[1,1]*x-w[1,2]*y-w[1,0]>=0 and testdata.label.iloc[i]==0))) :
                counter+=1            
    if (algo=='QDA'):
        mean0=gaussianMeanMLE(traindata,0)
        precision0=np.linalg.inv(QDA_Covariance(traindata,0))
        product0=np.dot(precision0,mean0)
        mean1=gaussianMeanMLE(traindata,1)
        precision1=np.linalg.inv(QDA_Covariance(traindata,1))
        product1=np.dot(precision1,mean1)
        bernoulli=bernoulliParameterMLE(traindata)
        for i in range(0,testdata.count()[0]):
            x=testdata.x.iloc[i]
            y=testdata.y.iloc[i]
            S1=precision0[0,0]*x**2+precision0[1,1]*y**2+2*x*y*precision0[0,1]-2*x*product0[0]-2*y*product0[1]+np.dot(mean0,product0)
            S2=precision1[0,0]*x**2+precision1[1,1]*y**2+2*x*y*precision1[0,1]-2*x*product1[0]-2*y*product1[1]+np.dot(mean1,product1)
            if((-0.5*S1+(0.5)*math.log2(np.linalg.det(precision0))+math.log2(1-bernoulli))-(-0.5*S2+(0.5)*math.log2(np.linalg.det(precision1))+math.log2(bernoulli))>=0 and testdata.label.iloc[i]==1) or ((-0.5*S1+(0.5)*math.log2(np.linalg.det(precision0))+math.log2(1-bernoulli))-(-0.5*S2+(0.5)*math.log2(np.linalg.det(precision1))+math.log2(bernoulli))<0 and testdata.label.iloc[i]==0):
                counter+=1           
    return counter/testdata.count()[0]*100

def QDA_Covariance(data,i):
    S=np.zeros((2,2))
    denominator=data[data.label==i].x.count()
    data_centred=(data[data.label==i][['x','y']]-gaussianMeanMLE(data,i))
    for j in range (0,len(data_centred)):
        S=S+np.dot(np.array(data_centred.iloc[j]).reshape(2,1),np.array(data_centred.iloc[j]).reshape(1,2))
    return S/denominator

def decision_function_0(data,x,y):
    mean=gaussianMeanMLE(data,0)
    precision=np.linalg.inv(QDA_Covariance(data,0))
    product=np.dot(precision,mean)  
    S1=precision[0,0]*x**2+precision[1,1]*y**2+2*x*y*precision[0,1]-2*x*product[0]-2*y*product[1]+np.dot(mean,product)
    return -0.5*S1+(0.5)*math.log2(np.linalg.det(precision))+math.log2(1-bernoulliParameterMLE(data))
