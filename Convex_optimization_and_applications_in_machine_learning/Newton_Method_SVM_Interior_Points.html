def phi(x,t,Q,p,A,b):
    self_concordant=-np.sum([log2(b[i]-np.dot(A[i][:],x)) for i in range(len(x))])
    return t*(0.5*np.dot(x,np.dot(Q,x))+np.dot(p,x))+self_concordant

def grad(x,t,Q,p,A,b):
    self_concordant_gradient=[np.sum([A[i][k]/(b[i]-np.dot(A[i][:],x)) for i in range(len(x))]) for k in range(len(x))]
    return [t*(np.dot(Q[i][:],x)+p[i])+self_concordant_gradient[i] for i in range(len(x))]
In [4]:
def hess(x,t,Q,p,A,b):
    self_concordant_hess=[[np.sum([(A[p][j]*A[p][i])/(b[p]-np.dot(A[p][:],x))**2 for p in range(len(x))]) for i in range(len(x))] for j in range(len(x))]
    return [[t*Q[i][j]+self_concordant_hess[i][j] for i in range(len(x))] for j in range(len(x))]
Please make sur that when calling those functions:
- x,p and b are lists of elements (i.e x=[1,2,3] or np.array([1,2,3]))
- Q and A are matrices (arrays) (i.e A=np.zeros((1,1)) or A=[[1]] when A is a scalar)).

def dampedNewtonStep(x,f,g,h):
    gradient=g(x)
    hes=h(x)
    square_newton_decrement=np.dot(gradient,np.dot(np.linalg.inv(hes),gradient))
    x_new=list(x-(1/(1+sqrt(square_newton_decrement)))*np.dot(np.linalg.inv(hes),gradient))
    gap=square_newton_decrement/2
    return [x_new,gap]

def dampedNewton(x0,f,g,h,tol):
    xhist=[x0]
    [x_new,gap]=dampedNewtonStep(x0,f,g,h)
    xhist.append(x_new)
    while(gap>tol):
        [x_new,gap]=dampedNewtonStep(x_new,f,g,h)
        xhist.append(x_new)    
    return [x_new,xhist]

def backtracking(x,f,g,direction,alpha,beta):
    t=1    # 0<alpha<0.5 and 0<beta<1
    f_x=f(x)
    gradient=g(x)
    while(f(np.array(x)+t*np.array(direction))>=f_x+alpha*t*np.dot(gradient,direction)):
        t=beta*t
    return t

def newtonStep(x,f,g,h):
    gradient=g(x)
    hes=h(x)
    square_newton_decrement=np.dot(gradient,np.dot(np.linalg.inv(hes),gradient))
    step=list(-np.dot(np.linalg.inv(hes),gradient))
    gap=square_newton_decrement/2
    return [step,gap]

def transform_svm_primal(tau,X,y):
    n,d=np.shape(X)[0],np.shape(X)[1]
    Q=np.zeros((n+d,n+d))
    Q[:d,:d]=np.eye(d)
    p1,p2=list(np.zeros(d)),list((1/(tau*n))*np.ones(n))
    p=np.array(p1+p2)
    b1,b2=list(-np.ones(n)),list(np.zeros(n))
    b=np.array(b1+b2)
    A=np.zeros((2*n,n+d))
    for i in range(n):
        A[i,:d]=-y[i]*X[i]
        A[i,d:]=list(-np.eye(n)[i])
        A[n+i,:d]=0
        A[n+i,d:]=list(-np.eye(n)[i])
    return [Q,p,A,b]

def transform_svm_dual(tau,X,y): 
    n,d=np.shape(X)[0],np.shape(X)[1]
    Q=np.array([[y[i]*y[j]*np.dot(X[i],X[j].T) for i in range(n)] for j in range(n)])
    p=-np.ones(n)
    A=np.zeros((2*n,n))
    b1,b2=list((1/(tau*n))*np.ones(n)),list(np.zeros(n))
    b=np.array(b1+b2)
    for i in range(n):
        A[i,:]=list(np.eye(n)[i])
        A[n+i,:]=list(-np.eye(n)[i])
    return [Q,p,A,b]

def barr_method(Q,p,A,b,x0,mu,tol):
    t=30
    f = lambda z: phi(z,t,Q,p,A,b)
    g = lambda z: grad(z,t,Q,p,A,b)
    h = lambda z: hess(z,t,Q,p,A,b)
    xhist=[x0]
    x_opt=dampedNewton(x0,f,g,h,tol)[0]
    xhist.append(x_opt)
    m=np.shape(A)[0]
    while(m>=t*tol):
        x_opt=dampedNewton(x_opt,f,g,h,tol)[0]
        xhist.append(x_opt)
        t=mu*t
    return[x_opt,xhist]

def duality_gap(Q,p,A,b,x0,mu,tol):
    duality_gap=[]
    t=1
    f = lambda z: phi(z,t,Q,p,A,b)
    g = lambda z: grad(z,t,Q,p,A,b)
    h = lambda z: hess(z,t,Q,p,A,b)
    xhist=[x0]
    x_opt=dampedNewton(x0,f,g,h,tol)[0]
    xhist.append(x_opt)
    m=np.shape(A)[0]
    duality_gap.append(m/t)
    itera=0
    while(m>=t*tol):
        x_opt=dampedNewton(x_opt,f,g,h,tol)[0]
        xhist.append(x_opt)
        t=mu*t
        duality_gap.append(m/t)
        itera+=1
    return [itera,duality_gap]
The duality_gap function returns the number of inner iterations until the while criteria is attained and all the duality gap values over those iterations in a list.

We can notice that the primal and dual problems give the same duality gap evolution, For mu=2 the condition of optimality approximation to respect is attained at the 14-th iteration while for mu=15 it's for 4-th iteration and 3-th iteration for mu=50 and mu=100, The width of each horizental step is the number of damped Newton steps calculated for that outer iteration. The height of each vertical step is equal to a multiple of mu because the duality gap is reduced by the factor mu at the end of each outer iteration.

The figure illustrates also the trade-off of the value mu, when mu is small the steps are short because the duality gap is reduced only by a factor of 2 per outer iteration. In contrast, mu=100 the steps are large since the duality gap is reduced only by a factor of 100 per outer iteration.
